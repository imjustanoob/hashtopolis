\section*{Preconfigured Tasks (\textit{pretask})}
		Used to access all functions around preconfigured tasks. 
		\subsection*{\textit{setPretaskPriority}}
			Set the priority of a preconfigured task.
			{
				\color{blue}
				\begin{verbatim}
				{
				  "section": "pretask",
				  "request": "setPretaskPriority",
				  "pretaskId": 1,
				  "priority": 100,
				  "accessKey": "mykey"
				}
				\end{verbatim}
			}
			{
				\color{OliveGreen}
				\begin{verbatim}
				{
				  "section": "pretask",
				  "request": "setPretaskPriority",
				  "response": "OK"
				}
				\end{verbatim}
			}
